    <script>
        function toggleBreak() {
            const section = document.getElementById('breakSection');
            const hasBreak = document.getElementById('hasBreak').checked;
            section.classList.toggle('hidden', !hasBreak);
            calculate();
        }

        function calculate() {
            const startVal = document.getElementById('startTime').value;
            const endVal = document.getElementById('endTime').value;
            
            if (!startVal || !endVal) return;

            let start = new Date(startVal);
            let end = new Date(endVal);

            if (end <= start) {
                document.getElementById('errorMessage').classList.remove('hidden');
                document.getElementById('resultArea').classList.add('hidden');
                return;
            }
            document.getElementById('errorMessage').classList.add('hidden');
            document.getElementById('resultArea').classList.remove('hidden');

            // 處理休息時間
            let breakMinutes = 0;
            const hasBreak = document.getElementById('hasBreak').checked;
            if (hasBreak) {
                const bStartVal = document.getElementById('breakStart').value;
                const bEndVal = document.getElementById('breakEnd').value;
                if (bStartVal && bEndVal) {
                    let bStart = new Date(bStartVal);
                    let bEnd = new Date(bEndVal);
                    if (bEnd > bStart && bStart >= start && bEnd <= end) {
                        breakMinutes = (bEnd - bStart) / 60000;
                    }
                }
            }

            // 計算每一分鐘的歸屬 (日間或夜間)
            let current = new Date(start);
            let dayMins = 0;
            let nightMins = 0;
            
            // 排除休息時段後的有效分鐘數分配
            const breakStart = hasBreak && document.getElementById('breakStart').value ? new Date(document.getElementById('breakStart').value) : null;
            const breakEnd = hasBreak && document.getElementById('breakEnd').value ? new Date(document.getElementById('breakEnd').value) : null;

            while (current < end) {
                // 檢查是否在休息時間內
                const isInBreak = breakStart && breakEnd && current >= breakStart && current < breakEnd;
                
                if (!isInBreak) {
                    const hour = current.getHours();
                    if (hour >= 6 && hour < 22) {
                        dayMins++;
                    } else {
                        nightMins++;
                    }
                }
                current.setMinutes(current.getMinutes() + 1);
            }

            const totalActiveMins = dayMins + nightMins;
            const totalActiveHours = totalActiveMins / 60;

            // 計算費用邏輯
            let totalFee = 0;
            
            // 比例計算 (假設比例分配在前2小時與後續時間)
            const dayRatio = dayMins / totalActiveMins;
            const nightRatio = nightMins / totalActiveMins;

            if (totalActiveHours <= 2) {
                // 前兩小時內
                totalFee = (dayMins / 60) * 1000 + (nightMins / 60) * 2000;
            } else {
                // 拆分前2小時與第3小時起
                const first2HourMins = 120;
                const after2HourMins = totalActiveMins - 120;

                // 前2小時費用 (按比例拆分日夜)
                const first2DayMins = 120 * dayRatio;
                const first2NightMins = 120 * nightRatio;
                totalFee += (first2DayMins / 60) * 1000;
                totalFee += (first2NightMins / 60) * 2000;

                // 第3小時起計費規則 (未滿30分以30分計，滿30分未滿1小時以1小時計)
                const after2HoursRaw = after2HourMins / 60;
                const fullHours = Math.floor(after2HoursRaw);
                const remainingMins = after2HourMins % 60;
                
                let billedAfter2Hours = fullHours;
                if (remainingMins > 0 && remainingMins < 30) {
                    billedAfter2Hours += 0.5;
                } else if (remainingMins >= 30) {
                    billedAfter2Hours += 1;
                }

                // 剩餘時間費用 (按比例拆分日夜)
                totalFee += (billedAfter2Hours * dayRatio) * 500;
                totalFee += (billedAfter2Hours * nightRatio) * 1000;
            }

            // 更新 UI
            document.getElementById('displayTotalTime').innerText = formatMins( (end - start) / 60000 );
            document.getElementById('displayBreakTime').innerText = breakMinutes > 0 ? formatMins(breakMinutes) : "無";
            document.getElementById('displayRatio').innerText = `日間 ${formatMins(dayMins)} / 夜間 ${formatMins(nightMins)}`;
            document.getElementById('displayTotalFee').innerText = `NT$ ${Math.round(totalFee).toLocaleString()}`;
        }

        function formatMins(mins) {
            const h = Math.floor(mins / 60);
            const m = Math.round(mins % 60);
            return `${h}時${m}分`;
        }
